      #!/bin/sh                            <br>
      INFO=$(tiffinfo "$1" 2>&1 | sed -r -n -e '/Warning/b' -e
      's/.*Tag 50839: *//p')<br>
                                                <br>
      doit() {                                  <br>
        node | xmlstarlet fo                    <br>
      }                                         <br>
                                                <br>
      doit << EOF                               <br>
        console.log(                            <br>
          new Buffer([$INFO])                   <br>
          .toString('utf8')                     <br>
          .replace(/\0|\n/g,'')                 <br>
          .replace(/^[^<]+<\?xml/,'<?xml')      <br>
          .replace(/\/properties>.*/,'/properties>')<br>
        );                                      <br>
      EOF                                       <br>
